	//随机取5张牌
	public function randarr($arr, $num) {
		$arrid = array_rand($arr, $num);
		foreach ($arrid as $v) {
			$newarr [] = $arr [$v];
		}
		return $newarr;
	}
	//php取出数组中的最大值
    public function getMax($arr) {
        $max = $arr[0];
        foreach ($arr as $k => $v) {
            if($v > $max){
                $max = $v;
            }
        }
        return $max;
    }
